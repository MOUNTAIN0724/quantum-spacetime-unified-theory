# v4.1定義
def beta_eff_v41(M):
    x = M / M_th
    if x < 1:
        f = exp(-(1-x))
    else:
        f = 1 - exp(-(x-1))
    return 0.279 * f

# v4.5.1定義（修正後）
def beta_eff_v451(M):
    x = M / M_th
    if x < 0.001:
        f = 0.001
    elif x < 0.01:      # 0.001 ≤ x < 0.01
        f = 0.01
    elif x < 0.1:       # 0.01 ≤ x < 0.1
        f = 0.1
    elif x < 0.5:       # 0.1 ≤ x < 0.5
        f = 0.5
    elif x < 0.8:       # 0.5 ≤ x < 0.8
        f = 0.7
    elif x < 1.0:       # 0.8 ≤ x < 1.0
        f = 0.7 + 0.1 * (x - 0.8) / 0.2
    elif x < 2.0:       # 1.0 ≤ x < 2.0
        f = 0.8 + 0.1 * (x - 1.0) / 1.0
    else:               # x ≥ 2.0
        f = 0.9
    return 0.8000 * f  # β₀從0.279變為0.8000
2. a_eff函數 (環境依賴加速度)
python
# v4.1：無此函數
# v4.5.1新增：
def effective_a0_ratio(sigma):
    A_low = 0.0150
    sigma_crit = 0.4000
    sigma_transition = 2.5000
    
    if sigma < sigma_crit:
        return A_low
    elif sigma < sigma_transition:
        frac = (sigma - sigma_crit) / (sigma_transition - sigma_crit)
        return A_low + (1.0 - A_low) * frac
    else:
        return 1.0

def galaxy_rotation_velocity(M_baryon, R_disk):
    # 計算表面密度σ
    sigma = M_baryon / (np.pi * R_disk**2)
    
    # 獲取環境依賴的a_eff
    a_ratio = effective_a0_ratio(sigma)
    a_eff = a0_standard * a_ratio
    
    # QST公式
    v_qst = (G * M_baryon * a_eff * conversion_factor)**0.25
    
    # 第五力修正
    beta_eff = beta_effective(M_baryon * 2e30)
    
    return v_qst * (1 + beta_eff)**0.25, a_ratio
