def beta_effective(M):
    """v4.5.1修正的β_eff函數"""
    beta0 = 0.8000
    M_th = 1.0e22
    x = M / M_th
    
    if x < 0.001:
        f = 0.001
    elif x < 0.01:      # 0.001 ≤ x < 0.01
        f = 0.01
    elif x < 0.1:       # 0.01 ≤ x < 0.1
        f = 0.1
    elif x < 0.5:       # 0.1 ≤ x < 0.5
        f = 0.5
    elif x < 0.8:       # 0.5 ≤ x < 0.8  (v4.5.1修復關鍵)
        f = 0.7
    elif x < 1.0:       # 0.8 ≤ x < 1.0
        f = 0.7 + 0.1 * (x - 0.8) / 0.2
    elif x < 2.0:       # 1.0 ≤ x < 2.0
        f = 0.8 + 0.1 * (x - 1.0) / 1.0
    else:               # x ≥ 2.0
        f = 0.9
    
    return beta0 * f

def effective_a0_ratio(sigma):
    """v4.5.1的a_eff函數"""
    A_low = 0.0150
    sigma_crit = 0.4000
    sigma_transition = 2.5000
    
    if sigma < sigma_crit:
        return A_low
    elif sigma < sigma_transition:
        # 線性插值
        frac = (sigma - sigma_crit) / (sigma_transition - sigma_crit)
        return A_low + (1.0 - A_low) * frac
    else:
        return 1.0

def dark_energy_density():
    """計算Ω_DE"""
    phi_plus = 1.621
    phi_minus = 1.459
    omega = 1.297
    m_phi = 0.08
    m_omega = 0.06
    mu = 0.00306
    V_const = 2.0527
    
    V_phi = 0.5 * m_phi**2 * (phi_plus**2 + phi_minus**2)
    V_mix = -mu**2 * phi_plus * phi_minus
    V_omega = 0.5 * m_omega**2 * omega**2
    V_total = V_phi + V_mix + V_omega + V_const
    
    rho_crit = 3.0  # 歸一化單位
    Omega_DE = V_total / rho_crit
    
    return Omega_DE  # 0.690309

def mars_time_delay():
    """計算火星時間延遲"""
    beta0 = 0.8000
    delta_phi_grav = 3.386e-9
    delta_tau_tau = beta0 * delta_phi_grav
    
    # 轉換為μs/日
    seconds_per_day = 86400.0
    microseconds_per_second = 1e6
    delta_tau_per_day = delta_tau_tau * seconds_per_day * microseconds_per_second
    
    return delta_tau_per_day  # 234.0 μs/日

def galaxy_rotation_velocity(M_baryon, R_disk):
    """
    計算星系旋轉速度
    
    參數:
        M_baryon: 重子質量 [M_☉]
        R_disk: 盤半徑 [kpc]
    
    返回:
        (V_QST, a_ratio): 旋轉速度 [km/s] 和 a_eff/a₀比例
    """
    import numpy as np
    
    # 常數
    G = 6.67430e-11     # m³/kg/s²
    M_SUN = 1.989e30    # kg
    KPC = 3.086e19      # m
    a0_standard = 1.2e-10  # m/s²
    
    # 計算表面密度
    M_baryon_kg = M_baryon * M_SUN
    R_disk_m = R_disk * KPC
    sigma = M_baryon_kg / (np.pi * R_disk_m**2)
    sigma = sigma / (1e9 * M_SUN / KPC**2)  # 轉換為 10⁹ M_☉/kpc²
    
    # 計算a_eff
    a_ratio = effective_a0_ratio(sigma)
    a_eff = a0_standard * a_ratio
    
    # 計算β_eff
    beta_eff = beta_effective(M_baryon_kg)
    
    # QST公式
    v4 = G * M_baryon_kg * a_eff * (1.0 + beta_eff) * 2.0
    v_qst = v4**0.25
    v_qst_km_s = v_qst / 1000.0
    
    return v_qst_km_s, a_ratio
